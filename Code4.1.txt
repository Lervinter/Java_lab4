import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class PalindromeCheckerTest {

    public static boolean isPalindrome(String word) {
        String cleanWord = word.replaceAll("\\s", "").toLowerCase();

        int length = cleanWord.length();
        for (int i = 0; i < length / 2; i++) {
            if (cleanWord.charAt(i) != cleanWord.charAt(length - 1 - i)) {
                return false;
            }
        }
        return true;
    }

    @Test
    void testIsPalindrome() {
        assertTrue(isPalindrome("level"));
        assertTrue(isPalindrome("A man a plan a canal Panama"));
        assertTrue(isPalindrome("Able was I ere I saw Elba"));
        
        assertFalse(isPalindrome("hello"));
        assertFalse(isPalindrome("world"));
        assertFalse(isPalindrome("Palindrome"));
        
        assertTrue(isPalindrome(""));
    }

    public static void main(String[] args) {
        org.junit.platform.console.ConsoleLauncher.execute(
                "--class-path", "out/production/your_project",
                "--scan-classpath"
        );
    }
}
